##### 1.防火墙相关

~~~shell
#查看防火墙运行状态
firewall-cmd --state

#操作防火墙服务
systemctl start firewalld # 启动服务
systemctl stop firewalld #关闭服务
systemctl restart firewalld #重启服务

#查看开放端口
firewall-cmd --list-ports

# 查看端口开放情况
firewall-cmd --query-port=80/tcp

#永久放行端口
firewall-cmd --add-port=80/tcp --permanent
# --permanent 表示永久放行
~~~
